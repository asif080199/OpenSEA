167 
168 
169  if( (&out != &A) && (&out != &B) )
170  {
171  out.set_size(A.n_rows, A.n_cols, A.n_slices + B.n_slices);
172 
173  out.slices(0, A.n_slices-1 ) = A;
174  out.slices(A.n_slices, out.n_slices-1) = B;
175  }
176  else // we have aliasing

181  C.slices(A.n_slices, C.n_slices-1) = B;
