18 trunc_log(const eT x)

21  {
22  if(x == std::numeric_limits<eT>::infinity())
23  {
24  return Math<eT>::log_max();
25  }
26  else
27  {
28  return (x <= eT(0)) ? Math<eT>::log_min() : std::log(x);

30  }

33  return std::log(x);
34  }
