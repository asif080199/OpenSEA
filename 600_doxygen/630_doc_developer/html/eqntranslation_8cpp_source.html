84 complex<double> EqnTranslation::setFormula()
85 {
86  complex<double> valOut;
87 
88  //I only want to define a single equation to linear translation. Add some if statements to change equation
89  //definition, depending on equation index.
90  if (this->getDataIndex() == 0) //Data index for X-translation
91  {
92  //Add in mass objects.
93  valOut = ForceMass(1) * Ddt(1,2) + //Direct mass object
94  ForceMass(5) * Ddt(5,2) +
95  ForceMass(6) * Ddt(6,2);
96  }
97  else if (this->getDataIndex() == 1) //Data index for Y-translation
98  {
99  //Add in mass objects.
100  valOut = ForceMass(2) * Ddt(2,2) +
101  ForceMass(4) * Ddt(4,2) +
102  ForceMass(6) * Ddt(5,2);
103  }
104  else if (this->getDataIndex() == 2) //Data index for Z-translation
105  {
106  //Add in mass objects.
107  valOut = ForceMass(3) * Ddt(3,2) +
108  ForceMass(4) * Ddt(4,2) +
109  ForceMass(5) * Ddt(5,2);
110  }
111 
112 
113  //Add in reactive force objects
114  //Sum hydrodynamic reactive forces for derivative orders 0 - 2
115  valOut += Sum("Func2()", "ord", 0, 2); //ForceReact_hydro
116 
117  //Sum user reactive forces for however high the derivative order goes.
118  valOut += Sum("Func4()", "ord"); //ForceReact_user
119 
120 
121  //Add in cross-body force objects
122  //Add cross-body forces for hydrodynamic forces. Sum for order from 0 to 2.
123  valOut += Sum("Func6()", "body"); //ForceCross_hydro
124  valOut += Sum("Func8()", "body"); //ForceCross_user
125 
126  //Add in active force objects.
127  //Active forces must be entered negative to account for the rearranged equation.
128  valOut -= ForceActive_user();
129  valOut -= ForceActive_hydro();
130 
131  //Write out results
132  return valOut;
133 }
