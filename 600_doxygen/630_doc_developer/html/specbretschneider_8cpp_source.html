56 double SpecBretschneider::getSpecEnergy(double freq)
57 {
58  //Variable declaration
59  double A; //The A expression in the equation.
60  double B; //The B expression in the equation.
61 
62  //Calculate A constant.
63  A = constA * pow(pHsig, 2) / pow(pTMean, 4);
64 
65  //Calculate B constant.
66  B = constB / pow(pTMean, 4);
67 
68  //Calculate output
69  return (A / pow(freq,5)) * exp(-B / pow(freq,4));
70 }
