353 {
354  //First assign the basic properties for the matbody.
355  listMatBody[bod].setId(bod);

365  MyModel->setBody(bod); //Set which body to use as the current body
366  MyModel->CoefficientOnly() = useCoeff; //Let it know to only calculate coefficients.
367  MyModel->Reset(); //Give it a reset just for good measure.
368 
369  //Iterate through all the active forces, user
370  //------------------------------------------
371  for (unsigned int i = 0; i < MyBod->listForceActive_user().size(); i++)
372  {

626 std::string getPath(string typePath)
627 {
628  std::string output; //Output of full path
629  char buff[1024]; //Buffer for reading path
630 
631  //First get executable path. Includes dependancy code for both windows and linux.
632  #ifdef Q_OS_WIN
633  //----------------------------------- Windows Code ----------------------------------------------------------------
634  ssize_t len = GetModuleFileName(NULL, buff, sizeof(buff) - 1);
635  std::string SLASH = "\\";
636  #elif defined Q_OS_LINUX
637  //------------------------------------ Linux Code -----------------------------------------------------------------
638  ssize_t len = readlink("/proc/self/exe", buff, sizeof(buff) - 1);
639  std::string SLASH = "/";
640  #endif
641 
642  //--------------------------- Platform Independent Code -----------------------------------------------------------
643  if (len != -1)
644  {
645  buff[len] = '\0';
646  output.assign(buff);
647  }
648  else
649  {
650  //Error handler later
651  }
652 
653  //Strip off name of executable itself.
654  std::string strip = SLASH + EXECNAME;
655  output.erase(output.size() - strip.size(), strip.size());
656 
657  //Strip back the name of the executable directory.
658  strip = BINFOLDER + SLASH + EXECFOLDER;
659  output.erase(output.size() - strip.size(), strip.size());
660 
661  //Change output depending on which path was requested
662  if (typePath == "exec")
663  {
664  //Add on the executable path
665  output.append(BINFOLDER);
666  output.append(SLASH);
667  output.append(EXECFOLDER);
668  }
669  else if (typePath == "bin")
670  {
671  //Add on the binary folder
672  output.append(BINFOLDER);
673  }
674  else if (typePath == "lib")
675  {
676  //Add on library path
677  output.append(LIBFOLDER);
678  }
679  else if (typePath == "var")
680  {
681  //Add on variable folder
682  output.append(VARFOLDER);
683  }
684  else if (typePath == "etc")
685  {
686  //Add on etc folder
687  output.append(ETCFOLDER);
688  }
689 
690  //Write output
691  return output;
692 }
