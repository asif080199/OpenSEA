424 {

436  MyModel->setBody(bod); //Set which body to use as the current body
437  MyModel->CoefficientOnly() = useCoeff; //Let it know to only calculate coefficients.
438  MyModel->Reset(); //Give it a reset just for good measure.
439 
440  //Iterate through all the active forces, user
441  //------------------------------------------
