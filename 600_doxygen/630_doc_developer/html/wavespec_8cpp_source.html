119 void WaveSpec::ConstructSpline(double freqIn)
120 {
121  //First find the index of the point closest to the requested frequency.
122  //Assumes the vectors are sequentially ordered.
123  double dist[2]; //Distances from frequency in vector to freqIn.
124  int index; //The index of the frequency with the closest match.
125  real_1d_array x; //x values for spline interpolation.
126  real_1d_array y; //y values for spline interpolation.
127 
128  //Put data into alglib required data type.
129  x.setcontent(pWaveFreq.size(), pWaveFreq.data());
130  y.setcontent(pWaveEnergy.size(), pWaveEnergy.data());
131 
132  dist[0] = 1E6;
133  dist[1] = 1E6;
134 
135  for (int i = 0; i < pWaveFreq.size(); i++)
136  {
137  //Iterate through and find the index of the frequency with the closest match.
138  dist[0] = abs(freqIn - pWaveFreq[i]);
139 
140  if (dist[0] < dist[1])
141  {
142  index = i;
143  dist[1] = dist[0];
144  }
145  }
146 
147  //Select the correct function to build the spline. Assumes the data has at least two data points.
148  if (pWaveFreq.size() == 2)
149  {
150  //Build linear spline.
151  alglib::spline1dbuildlinear(x, y, pSpecCurve);
152  }
153 
154  else if ((pWaveFreq.size() >= 5) &&
155  (index >= 2) &&
156  ((pWaveFreq.size() - 1 - index) >= 3))
157  {
158  //Check for Akima spline.
159  /* Requires 5 or more data points, and the points must be arranged in the correct order with at
160  * least two data points below the requested frequency, and three data points above the requested frequency.
161  */
162 
163  //Build Akima spline.
164  alglib::spline1dbuildakima(x, y, pSpecCurve);
165  }
166 
167  else if (pWaveFreq.size() >= 3)
168  {
169  //Check for cubic spline.
170 
171  //Build Hermite spline.
172  alglib::spline1dbuildcubic(x, y, pSpecCurve);
173  }
174 
175  //Notify that the curve is now built.
176  SplineReady = true;
177 }
