84 double SeaModel_LongCrest::getWaveAmp(double dirIn, int freqIndex)
85 {
86  //First calculate the wave energy for the requested direction and frequency.
87  double WaveEnergy = getWaveEnergy(dirIn, freqIndex);
88 
89  //Next get the frequency step for the defined frequency index.
90  double stepsize = getFreqStep(freqIndex);
91 
92  //Finally, calculate the wave amplitude
93  try
94  {
95  //Check for negative number.
96  if (WaveEnergy < 0)
97  {
98  std::string msg; //Error message.
99 
100  msg = string("Interpolated wave energy was negative for the requested wave direction and") +
101  string("frequency. Please check the sea model.\nWave Direction: ");
102  std::to_string(dirIn) + string(" rad\n") + string("Wave Frequency: ") +
103  std::to_string(plistWaveFreq[freqIndex]) + string(" rad/s");
104 
105  //Throw the error message.
106  throw std::runtime_error(msg);
107  }
108 
109 
110  //Calculate wave amplitude
111  return sqrt(2 * WaveEnergy * stepsize);
112  }
113  catch(std::exception &err)
114  {
115  logStd.Notify();
116  logErr.Write(string("Object: SeaModel_LongCrest, Function: getWaveAmp()\nError Message: ") + err.what());
117  }
118  catch(...)
119  {
120  logStd.Notify();
121  logErr.Write("Unknown error occurred. Object: SeaModel_LongCrest, Function: getWaveAmp()");
122  }
123 }
