78 int HydroReader::readHydroSys()
79 {
80  //First read the top level files
81  //Read environment file
82  readEnvironment();
83 
84  //Read directions file
85  readDirections();
86 
87  //Read frequencies file
88  readFrequencies();
89 
90  //Next Iterate through each of the wave directions and read those files.
91  for (WaveInd = 0; i < plistWaveDir.size(); WaveInd++)
92  {
93  //Read the hydro mass file
94  readHydroMass(i);
95 
96  //Read the hydro damping file
97  readHydroDamp(i);
98 
99  //Read the hydro stiffness file
100  readHydroStiff(i);
101 
102  //Read the cross mass file
103  readCrossMass(i);
104 
105  //Read the cross damping file
106  readCrossDamp(i);
107 
108  //Read the cross stiffness file
109  readCrossStiff(i);
110 
111  //Done reading in data. Transfer the data from temporary holding to final values.
112  //Each hydrodata object is its own hydrobody.
113  for (unsigned int j = 0; j < plistTempHydro.size(); j++)
114  {
115  //Find index of the hydrobody
116  int bod = findHydroManager(plistTempHydro[j].getHydroBodyName());
117 
118  if (bod < 0)
119  {
120  //No hydrobody found. Create a new one.
121  ptSystem->addHydroManager();
122  bod = ptSystem->listHydroManager().size();
123 
124  //Set the body name.
125  ptSystem->listHydroManager(bod).setHydroBodyName(
126  plistTempHydro[j].getHydroBodyName());
127 
128  //Set gravity for the body.
129  ptSystem->listHydroManager(bod).setGravity(pGravity);
130  }
131 
132  //Find the index of the hydrodata set that matches the wave amplitude.
133  int amp = findHydroDataAmp(bod,
134  plistTempHydro[j].getWaveAmp());
135 
136  if (amp < 0)
137  {
138  //No hydrodata set with that wave amplitude was found. Add the data set as a new one.
139  ptSystem->listHydroManager(bod).addHydroData(
140  plistTempHydro[j]);
141  }
142  else
143  {
144  //Hydrodata set was found. Add a new direction to that part of the list.
145  ptSystem->listHydroManager(bod).addHydroData(
146  amp,
147  plistTempHydro[j]);
148  }
149  }
150 
151  //Clear the list of hydrodata
152  plistTempHydro.clear();
153  }
154 }

473  }
