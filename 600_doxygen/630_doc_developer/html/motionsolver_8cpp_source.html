164 cx_mat *MotionSolver::sumDerivative(matForceReact *forceIn)
165 {
166  if (forceIn == NULL)
167  {
168  //No inputs. Return NULL
169  return NULL;
170  }
171 
172  else
173  {
174  //complex version of current wave frequency.
175  complexDouble waveFreq(this->curWaveFrequency, 0.0);
176  //Complex version of imaginary variable i.
177  complexDouble imagI(0.0, 1.0);
178  //complex scalar multiple. Will calculate later.
179  complexDouble scalarMult;
180 
181  cx_mat *output = new cx_mat;
182 
183  //resize output matrix
184  output->zeros(forceIn->getMatSize(), forceIn->getMatSize());
185 
186  //Iterate through each derivative order and add them together.
187  for (int i = 0 ; i <= forceIn->getMaxOrder() ; i++)
188  {
189  //Calculate scalar product.
190  scalarMult = pow(waveFreq, i) * pow(imagI, i);
191 
192  //Multiply through to derivative terms and add to total.
193  for (unsigned int row = 0 ; row < forceIn->getDerivative(i).n_rows ; row++)
194  {
195  for (unsigned int col = 0 ; col < forceIn->getDerivative(i).n_cols ; col++)
196  {
197  output->at(row,col) += scalarMult * forceIn->getDerivative(i).at(row,col);
198  }
199  }
200  }
201  //Write output
202  return output;
203  }
204 }

227

230

232  for (int j = 0 ; j <= forceIn[i]->getMaxOrder() ; j++)

260 void MotionSolver::calculateOutputs()
261 {
262  //create a temporary vector to hold outputs of summations - user forces
263  vector<matForceReact*> tempReactList_usr(plistBody.size());
264  vector<cx_mat*> ActiveList_usr(plistBody.size());
265  vector<matForceCross *> input(plistBody.size()); //Temporary vector for input into cross-body
266  vector < vector<matForceCross *> > tempCrossList_usr(plistBody.size(), input);
267 
268  //create a temporary vector to hold outputs of summations - hydro forces
269  vector<matForceReact*> tempReactList_hydro(plistBody.size());
270  vector<cx_mat*> ActiveList_hydro(plistBody.size());
271  vector<matForceCross *> input2(plistBody.size()); //Temporary vector for input into cross-body
272  vector < vector<matForceCross *> > tempCrossList_hydro(plistBody.size(), input2);
273 
274  //Sum forces for each body in the list.
275  for (curSumBody = 0; curSumBody < plistBody.size(); curSumBody++)
276  {
277  //Add user forces.
278  tempReactList_usr[curSumBody] = sumReactSet(plistBody[curSumBody].listForceReact_user());
279  ActiveList_usr[curSumBody] = sumActiveSet(plistBody[curSumBody].listForceActive_user());
280  tempCrossList_usr[curSumBody] = sumCrossSet(plistBody[curSumBody].listForceCross_user());
281 
282  //Add hydro forces.
283  tempReactList_hydro[curSumBody] = sumReactSet(plistBody[curSumBody].listForceReact_hydro());
284  ActiveList_hydro[curSumBody] = sumActiveSet(plistBody[curSumBody].listForceActive_hydro());
285  tempCrossList_hydro[curSumBody] = sumCrossSet(plistBody[curSumBody].listForceCross_hydro());
286 
287  //Create reactive force matrices for adding in mass
288  matForceReact *curReactSet;
289  matForceReact mass = plistBody[curSumBody].getForceReact_mass();
290 
291  //Assign reactive force matrices
292  if (tempReactList_usr[curSumBody] == NULL)
293  {
294  //Reactive force set has no forces defined. Create one.
295  curReactSet = new matForceReact();
296  tempReactList_usr[curSumBody] = curReactSet;
297  }
298  else
299  {
300  //Reactive force set has some forces defined. Get their current location.
301  curReactSet = tempReactList_usr[curSumBody];
302  }
303 
304  //Add the mass matrix
305  *curReactSet = *curReactSet + mass;
306  }
307  //Each set now reduced to a single force object, for each body.
308 
309  //Create temporary matrices to hold outputs of derivative summation - user forces
310  vector<cx_mat*> ReactList_usr;
311  vector< vector<cx_mat*> > CrossList_usr;
312  // No need to Sum for active forces. Already independent of derivative.
313 
314  //Create temporary matrices to hold outputs of derivative summation - hydro forces
315  vector<cx_mat*> ReactList_hydro;
316  // No need to Sum for active forces. Already independent of derivative.
317  vector< vector<cx_mat*> > CrossList_hydro;
318 
319  //Sum derivatives within each force.
320  for (curSumBody = 0; curSumBody < plistBody.size(); curSumBody++)
321  {
322  //Sum for user forces
323  ReactList_usr.push_back(sumDerivative(tempReactList_usr[curSumBody]));
324  CrossList_usr.push_back(sumDerivative(tempCrossList_usr[curSumBody]));
325 
326  //Sum for hydro forces
327  ReactList_hydro.push_back(sumDerivative(tempReactList_hydro[curSumBody]));
328  CrossList_hydro.push_back(sumDerivative(tempCrossList_hydro[curSumBody]));
329  }
330 
331  //Delete uneeded variables
332  for (curSumBody = 0; curSumBody < plistBody.size(); curSumBody++)
333  {
334  delete tempReactList_hydro[curSumBody];
335  delete tempReactList_usr[curSumBody];
336  delete input[curSumBody];
337  delete input2[curSumBody];
338 
339  for (unsigned int p = 0; p < tempCrossList_hydro[curSumBody].size(); p++)
340  delete tempCrossList_hydro[curSumBody][p];
341 
342  for (unsigned int p = 0; p < tempCrossList_usr[curSumBody].size(); p++)
343  delete tempCrossList_usr[curSumBody][p];
344  }
345  tempReactList_hydro.clear();
346  tempReactList_usr.clear();
347  input.clear();
348  input2.clear();
349  tempCrossList_hydro.clear();
350  tempCrossList_usr.clear();
351 
352  //Create temporary variables for total of all forces.
353  vector<cx_mat> ReactList;
354  vector<cx_mat> ActiveList;
355  vector< vector<cx_mat> > CrossList;
356  vector<cx_mat> tempCross;
357 
358  //Add different force types for each body object.
359  for (curSumBody = 0; curSumBody < plistBody.size(); curSumBody++)
360  {
361  //Add active force objects
362  ActiveList.push_back(
363  SumSingle(ActiveList_usr[curSumBody], ActiveList_hydro[curSumBody], -2)
364  );
365 
366  //Add reactive force objects
367  ReactList.push_back(
368  SumSingle(ReactList_usr[curSumBody], ReactList_hydro[curSumBody], -1)
369  );
370 
371  //Add cross-body force objects
372  for (unsigned int j = 0; j < plistBody.size(); j++)
373  {
374  tempCross.push_back(
375  SumSingle(CrossList_usr[curSumBody][j], CrossList_hydro[curSumBody][j], j)
376  );
377  }
378  CrossList.push_back( tempCross );
379  tempCross.clear();
380  }
381 
382  //Delete uneeded variables
383  for (curSumBody = 0; curSumBody < plistBody.size(); curSumBody++)
384  {
385  delete ReactList_hydro[curSumBody];
386  delete ReactList_usr[curSumBody];
387  delete ActiveList_hydro[curSumBody];
388  delete ActiveList_usr[curSumBody];
389 
390  for (unsigned int p = 0; p < CrossList_hydro[curSumBody].size(); p++)
391  delete CrossList_hydro[curSumBody][p];
392 
393  for (unsigned int p = 0; p < CrossList_usr[curSumBody].size(); p++)
394  delete CrossList_usr[curSumBody][p];
395  }
396  tempCross.clear();
397  ReactList_hydro.clear();
398  ReactList_usr.clear();
399  ActiveList_hydro.clear();
400  ActiveList_usr.clear();
401  CrossList_hydro.clear();
402  CrossList_usr.clear();
403 
404  //"Assemble Global Matrix"
405  vector<int> matStart;
406  vector<int> matEnd;
407  if(plistBody.size() == 1) //Single Body Matrix
408  {
409  //Dont include cross body forces if only 1 matrix and just use first index of vector
410  globReactiveMat = ReactList[0]; //A Matrix
411  globActiveMat = ActiveList[0]; //F Matrix
412 
413  //Create values for matStart and matEnd
414  matStart.push_back(0);
415  matEnd.push_back(ActiveList[0].n_rows - 1);
416  }
417  else //Multiple Bodies, must resize the matrices
418  {
419  int newMatrixSize = 0;
420 
421  for(unsigned int j = 0; j < plistBody.size(); j++)
422  {
423  //get the size of new matrix by summing all reactiveForceMatrices
424  newMatrixSize += ReactList[j].n_rows;
425 
426  //Add the sizes to get the starting position of each matrix
427  if (j == 0) {
428  matStart.push_back(0);
429  }
430  else
431  {
432  matStart.push_back(matStart[j-1] + ReactList[j].n_rows);
433  }
434 
435  //Add the sizes to get the ending positions of each matrix.
436  matEnd.push_back(matStart[j] + ReactList[j].n_rows - 1);
437  }
438 
439  //A Matrix
440  //resize matrix to include all bodies reactiveforces & cross Body Forces, matrix must be square (ex. 6x6)
441  //At the same time, fill matrix with zero values.
442  globReactiveMat.zeros(newMatrixSize, newMatrixSize);
443  //F Matrix
444  globActiveMat.zeros(newMatrixSize, 1); //column Matrix
445 
446  for (unsigned int j = 0; j < plistBody.size(); j++)
447  {
448  //Iterate through and insert the reactive force matrices for each body
449  globReactiveMat.submat(matStart[j], matStart[j], matEnd[j], matEnd[j]) = ReactList[j];
450 
451  //Iterate through and insert the active force matrices for each body
452  globActiveMat.submat(matStart[j], 0, matEnd[j], 0) = ActiveList[j];
453  }
454 
455  //Add in the cross body forces for each body.
456  //Iterate through each body.
457  for (unsigned int i = 0; i < plistBody.size(); i++)
458  {
459  //Iterate through the cross-body forces listed for each body
460  for(unsigned int j = 0; j < plistBody.size(); j++)
461  {
462  globReactiveMat.submat(matStart[i], matStart[j], matEnd[i], matEnd[j]) = CrossList[i][j];
463  }
464  }
465  }
466 
467  //Debug print out matrix values
468  DebugMatrix("Global Reactive Matrix", globReactiveMat);
469  DebugMatrix("Global Active Matrix", globActiveMat);
470 
471  //Solve for Unknown Matrix (the X Matrix) -- A*X=B where X is the unknown
472  try
473  {
474  globSolnMat = solve(globReactiveMat, globActiveMat, true); //true arg for more precise calculations
475  }
476  catch (int err)
477  {
478  //Insert error handler later.
479  }
480 
481  //Debug print out solution matrix
482  DebugMatrix("Global Solution Matrix", globSolnMat);
483 
484  //Split into vector of solutions, each represents per body
485  for(unsigned int i = 0; i < plistBody.size(); i++)
486  {
487  plistSolution.push_back(globSolnMat.submat(matStart[i], 0, matEnd[i], 0));
488  }
489 }

523 cx_mat MotionSolver::SumSingle(cx_mat *Input1, cx_mat *Input2, int ForceType)
524 {
525  //Create default matrix of zeros.
526  cx_mat output;
527  if (ForceType == -2)
528  {
529  //Active force matrix were the inputs. Create a column matrix.
530  //Get the dimensions for the output matrix.
531  int rows = plistBody[curSumBody].refMass().n_rows;
532  output.zeros(rows, 1);
533  }
534 
535  else if (ForceType == -1)
536  {
537  //Reactive force matrix. Create a square matrix.
538  int rows = plistBody[curSumBody].refMass().n_rows;
539  output.zeros(rows,rows);
540  }
541 
542  else
543  {
544  //Cross Body force matrix. The integer of the input specifies the second body index.
545  int rows = plistBody[curSumBody].refMass().n_rows;
546  int cols = plistBody[ForceType].refMass().n_rows;
547  output.zeros(rows,cols);
548  }
549 
550  //Add in inputs, depending on whether any of them are NULL values.
551  if (Input1 != NULL)
552  output = output + *Input1;
553 
554  if (Input2 != NULL)
555  output = output + *Input2;
556 
557  //Write output
558  return output;
