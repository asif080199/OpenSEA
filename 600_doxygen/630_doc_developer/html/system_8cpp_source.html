578 void System::linkBodies(int bodID)
579 {
580  //Iterate through all user linked bodies
581  for (unsigned int i = 0; i < plistBody[bodID].listNamedLink_user().size(); i++)
582  {
583  //Search for body with correct name
584  string name = plistBody[bodID].listNamedLink_user(i); //Name of the body to search for.
585 
586  for (unsigned int j = 0; j < listBody().size(); j++)
587  {
588  if (name == listBody(j).refBodyName())
589  {
590  //Assign the pointer for the body
591  plistBody[bodID].listCrossBody_user(i) = listBody(j);
592  //Quit the loop
593  break;
594  }
595  }
596  }
597  //Clear the list of named links. Don't need it anymore.
598  plistBody[bodID].listNamedLink_user().clear();
599 
600  //Iterate through all hydro linked bodies
601  for (unsigned int i = 0; i < plistBody[bodID].listNamedLink_hydro().size(); i++)
602  {
603  //Search for body with correct name
604  string name = plistBody[bodID].listNamedLink_hydro(i); //Name of the body to search for.
605 
606  for (unsigned int j = 0; j < listBody().size(); j++)
607  {
608  if (name == listBody(j).refBodyName())
609  {
610  //Assign the pointer for the body
611  plistBody[bodID].listCrossBody_hydro(i) = listBody(j);
612  //Quit the loop
613  break;
614  }
615  }
616  }
617  //Clear the list of named links. Don't need it anymore.
618  plistBody[bodID].listNamedLink_hydro().clear();
619 }
