402 void System::linkBodies(int bodID)
403 {
404  //Iterate through all user linked bodies
405  for (unsigned int i = 0; i < plistBody[bodID].listNamedLink_user().size(); i++)
406  {
407  //Search for body with correct name
408  string name = plistBody[bodID].listNamedLink_user(i); //Name of the body to search for.
409 
410  for (unsigned int j = 0; j < listBody().size(); j++)
411  {
412  if (name == listBody(j).refBodyName())
413  {
414  //Assign the pointer for the body
415  plistBody[bodID].listCrossBody_user(i) = listBody(j);
416  //Quit the loop
417  break;
418  }
419  }
420  }
421  //Clear the list of named links. Don't need it anymore.
422  plistBody[bodID].listNamedLink_user().clear();
423 
424  //Iterate through all hydro linked bodies
425  for (unsigned int i = 0; i < plistBody[bodID].listNamedLink_hydro().size(); i++)
426  {
427  //Search for body with correct name
428  string name = plistBody[bodID].listNamedLink_hydro(i); //Name of the body to search for.
429 
430  for (unsigned int j = 0; j < listBody().size(); j++)
431  {
432  if (name == listBody(j).refBodyName())
433  {
434  //Assign the pointer for the body
435  plistBody[bodID].listCrossBody_hydro(i) = listBody(j);
436  //Quit the loop
437  break;
438  }
439  }
440  }
441  //Clear the list of named links. Don't need it anymore.
442  plistBody[bodID].listNamedLink_hydro().clear();
443 }
