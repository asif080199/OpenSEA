624 int dictSeaEnv::defineClass(string nameIn)
625 {
626  if (nameIn == OBJECT_WAVE_SPEC)
627  {
628  //Start creation of a wave spectrum object
629  pClassType = nameIn;
630  pClassSet = false;
631  return 0;
632  }
633 
634  else if (nameIn == OBJECT_SEA_MODEL)
635  {
636  //Start creation of a sea model object
637  pClassType = nameIn;
638  pClassSet = false;
639  return 0;
640  }
641 
642  else if (nameIn == OBJECT_WAVE_DEF)
643  {
644  //Start definition of a wave definition within a sea model.
645  //Notify that a new object needs to be built.
646  pMakeNewWaveDef = true;
647  return 0;
648  }
649 
650  else
651  {
652  //Case if no match found.
653  logStd.Notify();
654  logErr.Write(string("No class name found to match the input. Class specified: ") + nameIn);
655  return 1;
656  }
