103 void Derivative::addModelEquation(std::vector<double> listCoeffsIn, int EqnDataIn)
104 {
105  //Adds a new equation object to the list for the current derivative.
106  Equation newEqn;
107 
108  //Set data index for new equation.
109  if (EqnDataIn < 0)
110  {
111  //No data index assigned. Create a default. Default is the last item in the list.
112  //List will be expanded by 1 when I add the new equation object in. So I take the size of the list.
113  EqnDataIn = pEquationList.size();
114  }
115  newEqn.setDataIndex(EqnDataIn);
116 
117  //Iterate through and for each object in the list of coefficients, add to the equation model.
118  //Only add the object if it is non zero
119  for (unsigned int i = 0; i < listCoeffsIn.size(); i++)
120  {
121  if (listCoeffsIn[i] != 0.0)
122  {
123  //Add each coefficient to the list of coefficients for the equation object.
124  newEqn.addVariable(listCoeffsIn[i], i);
125  }
126  }
127 
128  //Finally add the new equation object to the list of equation objects.
129  pEquationList.push_back(newEqn);
130 }

139 int Derivative::findIndex(int indexIn)
140 {
141  int output; //integer to write as output.
142  int check; //The integer to check against.
143  bool test = false; //Boolean to check if a match was found.
144 
145  //Finds the integer of the equation object by data index.

148  //Check the data index of the object.

150  {
151  //No data index set. Use the position in the list.
152  check = i;
153  }
154  else

158  }
159

167  }
