61 double SpecJONSWAP::getSpecEnergy(double freq)
62 {
63  //Variable declaration
64  double A; //The A expression in the equation.
65  double B; //The B expression in the equation.
66  double C; //The C expression in the equation. THis creates the peakedness of the spectrum.
67  double J; //The J expression. This is the exponent applied in the C parameter.
68  double output; //The calculated output.
69  double gamexp; //The value for the gamma exponent expression.
70 
71  //Calculate A constant.
72  A = constA * pow(pHsig, 2) / pow(pTMean, 4);
73 
74  //Calculate B constant.
75  B = constB / pow(pTMean, 4);
76 
77  //Calculate output, for Bretschneider spectrum
78  output = (A / pow(freq,5)) * exp(-B / pow(freq,4));
79 
80  //Calculate gamma exponent
81  if (freq < 2 * PI / pTMean)
82  gamexp = GAM1;
83  else
84  gamexp = GAM2;
85 
86  //Calculate J value
87  J = exp( (-1 / (2 * pow(gamexp, 2))) * pow((freq * pTMean / (2 * PI)) - 1, 2) );
88 
89  //Calculate C value
90  C = pow(pGamma, J);
91 
92  //Calculate output
93  return constParam * C * output;
94 }
