105 complex<double> Dictionary::convertComplex(string input)
106 {
107  complex<double> output;
108 
109  const string BRACKET = "<";
110  const string IMAGINARY = "i";
111  const string PLUS = "+";
112  const string MINUS = "-";
113 
114  //First check if defined by amplitude and angle
115  int indexSep;
116  indexSep = input.find(BRACKET);
117  if (indexSep == std::string::npos)
118  {
119  //Not using bracket
120  int indPlus;
121  int indMinus;
122  int indImag;
123 
124  //Get positions of plus, minus, and imaginary.
125  //Ignore the first character to avoid picking up any plus/minus indicators for real component.
126  indImag = input.find(IMAGINARY,1);
127  indPlus = input.find(PLUS,1);
128  indMinus = input.find(MINUS,1);
129 
130  //Separate string by position of plus and minus signs.
131  if (indPlus == std::string::npos)
132  {
133  //No plus involved. Minus sign.
134  //Check if imaginary symbol is at middle or end of input.
135  if (indImag == input.length() - 1)
136  {
137  //End of input
138  //Assign real part.
139  output.real(
140  atof(input.substr(0, indMinus + 1).c_str())
141  );
142  //Assign imaginary part
143  output.imag(
144  -1.0 * atof(input.substr(indMinus + 1, indImag - indPlus).c_str())
145  );
146  }
147  else
148  {
149  //Middle of input
150  //Assign real part.
151  output.real(
152  atof(input.substr(0, indMinus).c_str())
153  );
154  //Assign imaginary part
155  output.imag(
156  -1.0 * atof(input.substr(indImag + 1, input.length() - 1 - indImag).c_str())
157  );
158  }
159  }
160  else
161  {
162  //Plus sign used.
163  //Check if imaginary symbol is at middle or end of input.
164  if (indImag == input.length() - 1)
165  {
166  //End of input.
167  //Assign real part.
168  output.real(
169  atof(input.substr(0, indPlus).c_str())
170  );
171  //Assign imaginary part
172  output.imag(
173  atof(input.substr(indPlus + 1, indImag - indPlus).c_str())
174  );
175  }
176  else
177  {
178  //Middle of input.
179  //Assign real part.
180  output.real(
181  atof(input.substr(0,indPlus).c_str())
182  );
183  //Assign imaginary part
184  output.imag(
185  atof(input.substr(indImag + 1, input.length() - 1 - indImag).c_str())
186  );
187  }
188  }
189  }
190  else
191  {
192  //using bracket. Process as polar notation.
193  double amp;
194  double phase;
195 
196  //Get amplitude
197  amp = atof(input.substr(0, indexSep).c_str());
198  //Get phase
199  phase = atof(input.substr(indexSep + 1).c_str());
200 
201  //Convert to real and imaginary parts.
202  output.real(amp * cos(phase));
203  output.imag(amp * sin(phase));
204  }
205 
206  //Return output
207  return output;
208 }
