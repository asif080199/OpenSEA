151 complex<double> Dictionary::convertComplex(string input)
152 {
153  complex<double> output;
154 
155  const string BRACKET = "<";
156  const string IMAGINARY = "i";
157  const string PLUS = "+";
158  const string MINUS = "-";
159  const string COMMA = ",";
160 
161  //First check if defined by amplitude and angle
162  int indexPolar;
163  indexPolar = input.find(BRACKET);
164 
165  //Next check if defined by comma.
166  int indexCoord;
167  indexCoord = input.find(COMMA);
168 
169  //Next check if defined by rectangular notation.
170  int indImag;
171  indImag = input.find(IMAGINARY,1);
172 
173  try
174  {
175 
176  if (indexPolar != std::string::npos)
177  {
178  //using bracket. Process as polar notation.
179  double amp;
180  double phase;
181 
182  //Get amplitude
183  amp = atof(input.substr(0, indexPolar).c_str());
184  //Get phase
185  phase = atof(input.substr(indexPolar + 1).c_str());
186 
187  //Convert to real and imaginary parts.
188  output.real(amp * cos(phase));
189  output.imag(amp * sin(phase));
190  }
191 
192  else if (indexCoord != std::string::npos)
193  {
194  //Using comma. Process as coordinate system notation.
195 
196  //Process real part.
197  output.real(
198  atof(input.substr(0,indexCoord).c_str())
199  );
200 
201  //Process imaginary part.
202  output.imag(
203  atof(input.substr(indexCoord + 1).c_str())
204  );
205  }
206 
207  else if (indImag != std::string::npos)
208  {
209  //using rectangular notation
210  int indPlus;
211  int indMinus;
212 
213 
214  //Get positions of plus, minus, and imaginary.
215  //Ignore the first character to avoid picking up any plus/minus indicators for real component.
216  indPlus = input.find(PLUS,1);
217  indMinus = input.find(MINUS,1);
218 
219  //Separate string by position of plus and minus signs.
220  if (indPlus == std::string::npos)
221  {
222  //No plus involved. Minus sign.
223  //Check if imaginary symbol is at middle or end of input.
224  if (indImag == input.length() - 1)
225  {
226  //End of input
227  //Assign real part.
228  output.real(
229  atof(input.substr(0, indMinus + 1).c_str())
230  );
231  //Assign imaginary part
232  output.imag(
233  -1.0 * atof(input.substr(indMinus + 1, indImag - indPlus).c_str())
234  );
235  }
236  else
237  {
238  //Middle of input
239  //Assign real part.
240  output.real(
241  atof(input.substr(0, indMinus).c_str())
242  );
243  //Assign imaginary part
244  output.imag(
245  -1.0 * atof(input.substr(indImag + 1, input.length() - 1 - indImag).c_str())
246  );
247  }
248  }
249  else
250  {
251  //Plus sign used.
252  //Check if imaginary symbol is at middle or end of input.
253  if (indImag == input.length() - 1)
254  {
255  //End of input.
256  //Assign real part.
257  output.real(
258  atof(input.substr(0, indPlus).c_str())
259  );
260  //Assign imaginary part
261  output.imag(
262  atof(input.substr(indPlus + 1, indImag - indPlus).c_str())
263  );
264  }
265  else
266  {
267  //Middle of input.
268  //Assign real part.
269  output.real(
270  atof(input.substr(0,indPlus).c_str())
271  );
272  //Assign imaginary part
273  output.imag(
274  atof(input.substr(indImag + 1, input.length() - 1 - indImag).c_str())
275  );
276  }
277  }
278  }
279 
280  else
281  throw std::runtime_error(string("Complex variable format not recognized. \n") +
282  string("Offending Entry: ") + input);
283  }
284  catch(std::exception &err)
285  {
286  logStd.Notify();
287  logErr.Write(string("Object: Dictionary, Function: convertComplex\n") + err.what());
288  exit(1);
289  }
290  catch(...)
291  {
292  logStd.Notify();
293  logErr.Write(string("Unknown error occurred. Object: Dictionary, Function: convertComplex"));
294  exit(1);
295  }
296 
297  //Return output
298  return output;
299 }
