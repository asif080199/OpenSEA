414 int FileWriter::writeGlobalMotion()
415 {
416  string writeFilename; //The name of the file to write out.
417  string classname; //The name of the class type that contains the file elements.
418  int errVal = 0; //Returned value for error codes from calculations. Default set to no error.
419  bool outputAvail = false; //Boolean to track if any output objects are even available.
420 
421  logStd.Write("\tBody motions");
422 
423  //Check if output objects even exist.
424  if (pOutput->listGlobalMotion().size() == 0)
425  outputAvail = false;
426  else
427  outputAvail = true;
428 
429  //create directory. Function will do nothing if directory already exists.
430  createDir(getCurWaveDir());
431 
432  //create filename and file output stream.
433  writeFilename = projectDirectory + SLASH + getCurWaveDir() + FILE_GLOBAL_MOTION;
434  ofstream output;
435 
436  QString filename(writeFilename.c_str());
437  QFile file(filename);
438 
439  //Check if file already exists.
440  if (file.exists())
441  {
442  //Already exists. Just open the file.
443  //Open file
444  output.open(writeFilename.c_str());
445  }
446  else
447  {
448  //Didn't already exist. Create a new file and add the header information.
449  //create file
450  output.open(writeFilename.c_str());
451 
452  //Add top header
453  output << header;
454 
455  if (outputAvail)
456  {
457  //Get the name for the class written in output.)
458  classname = pOutput->listGlobalMotion(0).getClassName();
459 
460  //Write the file info block.
461  output << getInfoBlock(classname);
462  output << BREAK_TOP;
463  }
464  }
465 
466  if (outputAvail)
467  {
468  //Write output for beginning of body
469  output << KEY_BODY << SPACE << OBJECT_BEGIN << endl;
470  output << TAB() << KEY_NAME << SPACE << QUOTE << pOutput->refCurBody().refBodyName() << QUOTE << END << endl;
471 
472  //Repeat process for each item in object list
473  for (unsigned int i = 0; i < pOutput->listGlobalMotion().size(); i++)
474  {
475  try
476  {
477  //Calculate the results for the given body and specified output object.
478  errVal = pOutput->calcGlobalMotion(i);
479 
480  //Check for errors
481  if (errVal != 0)
482  throw std::runtime_error("Error calculating derived output: body motion");
483 
484  //Start the output object.
485  output << TAB() << classname << SPACE << OBJECT_BEGIN << endl;
486  output << TAB() << KEY_NAME << SPACE << QUOTE << pOutput->listGlobalMotion(i).getName() << QUOTE;
487  output << END << endl;
488 
489  //Iterate for each frequency in the list
490  for (unsigned int j = 0; j < pOutput->listFreq().size(); j++)
491  {
492  //Create data signifier
493  output << TAB(2) << KEY_DATA << SPACE << OBJECT_BEGIN << endl;
494  //Add frequency designator
495  output << TAB(3) << KEY_FREQUENCY << SPACE << (j+1) << END << endl;
496  //Add value indicator
497  output << TAB(3) << KEY_VALUE << SPACE << LIST_BEGIN << endl;
498 
499  //Write out data for given frequency.
500  for (unsigned int k = 0; k < pOutput->listResult(j).n_rows; k++)
501  {
502  //Set precision
503  output.precision(DIGIT);
504  //Write output values for each solution object.
505  //Write the real part
506  output << TAB(4) << std::setprecision(DIGIT) << pOutput->listResult(j)(k,0).real();
507 
508  //Write the imaginary part
509  //Set precision
510  if (pOutput->listResult(j)(k,0).imag() < 0.00)
511  output << std::setprecision(DIGIT) << pOutput->listResult(j)(k,0).imag() << "i" << endl;
512  else
513  {
514  output << "+";
515  output << std::setprecision(DIGIT) << pOutput->listResult(j)(k,0).imag();
516  output << "i" << endl;
517  }
518  }
519  //Close list
520  output << TAB(3) << LIST_END << END << endl;
521  //Close data object
522  output << TAB(2) << OBJECT_END << endl;
523  }
524  }
525  catch(std::exception &err)
526  {
527  logErr.Write("Unknown Error Occurred. Object: FileWriter, Function: writeGlobalMotion()\n" +
528  string("Error Message: ") + string(err.what()));
529  logStd.Notify();
530  errVal = 1;
531  }
532  catch(...)
533  {
534  //Write out error message
535  logErr.Write("Uknown error occurred. Ojbect: filewriter(), Function: writeGlobalMotion()");
536  logStd.Notify();
537  errVal = 1;
538  }
539 
540  //End the output object
541  output << TAB() << OBJECT_END << EOL;
542  }
543 
544  //Write output for ending of body
545  output << OBJECT_END << EOL << BREAK_BOTTOM << EOL;
546  }
547 
548  //Close file
549  output.close();
550 
551  //Report status of efforts
552  return errVal;
553 }
