176 int Equation::findIndex(int DataIndexIn)
177 {
178  int output; //integer to write as output.
179  int check; //The integer to check against.
180  bool test = false; //Boolean to tell if a match was found
181 
182  //Finds the integer of the equation object by data index.
183  for (unsigned int i = 0 ; i < plistDataIndex.size() ; i++)
184  {
185  //Check the data index of the object.
186  if (plistDataIndex[i] < 0)
187  {
188  //No data index set. Use the position in the list.
189  check = i;
190  }
191  else
192  {
193  //Data index is used. Use the position in the list.
194  check = plistDataIndex[i];
195  }
196 
197  //Check for match
198  if (check == DataIndexIn)
199  {
200  output = i;
201  test = true;
202  break;
203  }
204  }
205 
206  //Check for a match
207  if (!test)
208  {
209  //No match. Throw an exception
210  throw 1;
211  }
212 
213  return output;
214 }
