183 int Equation::findIndex(int DataIndexIn)
184 {
185  int output; //integer to write as output.
186  int check; //The integer to check against.
187  bool test = false; //Boolean to tell if a match was found
188 
189  //Finds the integer of the equation object by data index.
190  for (unsigned int i = 0 ; i < plistDataIndex.size() ; i++)
191  {
192  //Check the data index of the object.
193  if (plistDataIndex[i] < 0)
194  {
195  //No data index set. Use the position in the list.
196  check = i;
197  }
198  else
199  {
200  //Data index is used. Use the position in the list.
201  check = plistDataIndex[i];
202  }
203 
204  //Check for match
205  if (check == DataIndexIn)
206  {
207  output = i;
208  test = true;
209  break;
210  }
211  }
212 
213  //Check for a match
214  if (!test)
215  {
216  //No match. Throw an exception
217  throw 1;
218  }
219 
220  return output;
221 }
