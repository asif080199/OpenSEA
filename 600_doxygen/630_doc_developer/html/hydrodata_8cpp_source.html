231 matForceCross hydroData::getDataCross(double freqIn, int hydroInd)
232 {
233  double dist[2]; //The distance between the search term and terms considered.
234  vector<int> output(2); //The vector of output terms.
235 
236  //Set initial values
237  dist[0] = 1E9;
238  dist[1] = 1E9;
239 
240  //Iterate through data set and find the closest match
241  for (int i = 0; i < plistWaveFreq.size(); i++)
242  {
243  //Calculate distance
244  dist[0] = fabs(freqIn - plistWaveFreq[i]);
245 
246  //Check if distance is smaller
247  //And make sure that vector is large enough for specified index.
248  if ((dist[0] < dist[1]) &&
249  (plistDataCross[i].size() >= hydroInd + 1))
250  {
251  dist[1] = dist[0];
252  output[0] = i;
253  }
254  }
255 
256  //Reset distance values
257  dist[0] = 1E9;
258  dist[1] = 1E9;
259 
260  //Iterate through and find the 2nd closest match
261  for (int i = 0; i < plistWaveFreq.size(); i++)
262  {
263  //Calculate distance
264  dist[0] = fabs(freqIn - plistWaveFreq[i]);
265 
266  //Check if distance is smaller and not the other index.
267  //Also be sure that vector is large enough for the specified index.
268  if ((dist[0] < dist[1]) &&
269  (i != output[0]) &&
270  (plistDataCross[i].size() >= hydroInd + 1))
271  {
272  dist[1] = dist[0];
273  output[1] = i;
274  }
275  }
276 
277  //Perform interpolation
278  return iPolate(freqIn,
279  plistWaveFreq[output[0]],
280  plistWaveFreq[output[1]],
281  plistDataCross[output[0]][hydroInd],
282  plistDataCross[output[1]][hydroInd])
283  ;
284 }

287 matForceCross hydroData::getDataCross(double freqIn, std::string hydroName)
288 {
289  double dist[2]; //The distance between the search term and terms considered.
290  vector<int> output(2); //The vector of output terms.
291 
292  //Set initial values
293  dist[0] = 1E9;
294  dist[1] = 1E9;
295 
296  //Iterate through data set and find the closest match
297  //Include requirement that searched terms must match hydroName.
298  for (int i = 0; i < plistWaveFreq.size(); i++)
299  {
300  for (int j = 0; j < plistDataCross[i].size(); j++)
301  {
302  //Calculate distance
303  dist[0] = fabs(freqIn - plistWaveFreq[i]);
304 
305  //Check if distance is smaller
306  //And require that frequency contains a hydrobody with the matching name.
307  if ((dist[0] < dist[1]) &&
308  (plistDataCross[i][j].getLinkedName().compare(hydroName) == 0))
309  {
310  dist[1] = dist[0];
311  output[0] = i;
312  }
313  }
314  }
315 
316  //Reset distance values
317  dist[0] = 1E9;
318  dist[1] = 1E9;
319 
320  //Iterate through and find the 2nd closest match
321  for (int i = 0; i < plistWaveFreq.size(); i++)
322  {
323  for (int j = 0; j < plistDataCross[i].size(); j++)
324  {
325  //Calculate distance
326  dist[0] = fabs(freqIn - plistWaveFreq[i]);
327 
328  //Check if distance is smaller and not the other index.
329  //And require that frequency contains a hydrobody with the matching name.
330  if ((dist[0] < dist[1]) &&
331  (i != output[0]) &&
332  (plistDataCross[i][j].getLinkedName().compare(hydroName) == 0))
333  {
334  dist[1] = dist[0];
335  output[1] = i;
336  }
337  }
338  }
339 
340  //Now find the indices for the actual hydrobody names.
341  int hydroInd[2]; //Index of hydrobody names
342  for (int i = 0; i < 2; i++)
343  {
344  for (int j = 0; j < plistDataCross[i].size(); j++)
345  {
346  if (plistDataCross[i][j].getLinkedName().compare(hydroName) == 0)
347  {
348  hydroInd[i] = j;
349  break;
350  }
351  }
352  }
353 
354  //Perform interpolation
355  return iPolate(freqIn,
356  plistWaveFreq[output[0]],
357  plistWaveFreq[output[1]],
358  plistDataCross[output[0]][hydroInd[0]],
359  plistDataCross[output[1]][hydroInd[1]]
360  );
361 }

364 hydroData hydroData::interpHydroData(double freqIn)
365 {
366  //Create output object.
367  hydroData output;
368 
369  //Reproduce the constants.
370  output.pWaveAmp = this->pWaveAmp;
371  output.pWaveDir = this->pWaveDir;
372  output.pHydroBodyName = this->pHydroBodyName;
373 
374  //Add the single wave frequency to the output data list.
375  output.addWaveFreq(freqIn);
376 
377  //Interpolate for the active force, and add to the list.
378  output.addDataActive(
379  this->getDataActive(freqIn));
380 
381  //Interpolate for the reactive force, and add to the list.
382  output.addDataReact(
383  this->getDataReact(freqIn));
384 
385  //Iterate through each item in the data cross. Interpolate and add to the list.
386  for (int i = 0; i < plistDataCross.at(0).size(); i++)
387  {
388  output.addDataCross(
389  this->getDataCross(freqIn, i),
390  0);
391  }
392 
393  //Return output.
394  return output;
395 }
