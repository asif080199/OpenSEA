60 double SeaModel_SingleDirection::getWaveEnergy(double dirIn, double freqIn)
61 {
62  //First calculate wave spectrum energy
63  double SpecEnergy = listWaveSpec(0)->getSpecEnergy(freqIn);
64 
65  /*Do not need to check on wave scaling properties. The inheritted functions automatically
66  *ensure that any added wave spectrum has a wave scaling factor of 1 added by default.
67  */
68 
69  //Now scale wave energy based on wave direction.
70  //Write result.
71  return (2 / PI) * pow(

73  2) *
74  listWaveScale(0) *

79 double SeaModel_SingleDirection::getWaveAmp(double dirIn, int freqIndex)
80 {
81  //First calculate the wave energy for the requested direction and frequency.
82  double WaveEnergy = getWaveEnergy(dirIn, freqIndex);
83 
84  //Next get the frequency step for the defined frequency index.
85  double stepsize = getFreqStep(freqIndex);
86 
87  //Finally, calculate the wave amplitude
88  try
89  {
90  //Check for negative number.
91  if (WaveEnergy < 0)
92  {
93  std::string msg; //Error message.
94 
95  msg = string("Interpolated wave energy was negative for the requested wave direction and") +
96  string("frequency. Please check the sea model.\nWave Direction: ");
97  std::to_string(dirIn) + string(" rad\n") + string("Wave Frequency: ") +
98  std::to_string(plistWaveFreq[freqIndex]) + string(" rad/s");
99 
100  //Throw the error message.
101  throw std::runtime_error(msg);
102  }
103 
104 
105  //Calculate wave amplitude
106  return sqrt(2 * WaveEnergy * stepsize);
107  }
108  catch(std::exception &err)
109  {
110  logStd.Notify();
111  logErr.Write(string("Object: SeaModel_SingleDirection, Function: getWaveAmp()\nError Message: ") + err.what());
112  }
113  catch(...)
114  {
115  logStd.Notify();
116  logErr.Write("Unknown error occurred. Object: SeaModel_SingleDirection, Function: getWaveAmp()");
117  }
118 }
