83 std::complex<double> EqnRotation::setFormula()
84 {
85  std::complex<double> valOut;
86 
87  //Add in mass objects
88  valOut = Sum("Func1()", "var"); //ForceMass
89 
90  //Add in reactive force objects
91  valOut += Sum("Func10()", "ord", 0, 2); //ForceReact_hydro
92  valOut += Sum("Func11()", "ord"); //ForceReact_user
93 
94  //Add in cross-body force objects
95  valOut += Sum("Func6()", "body"); //ForceCross_hydro
96  valOut += Sum("Func9()", "body"); //ForceCross_user
97 
98  //Add in active force objects.
99  //Active forces must be entered negative to account for the rearranged equation.
100  valOut -= ForceActive_user(); //ForceActive_user
101  valOut -= ForceActive_hydro(); //ForceActive_hydro
102 
103  //Write out results
104  return valOut;
105 }
